\RequirePackage{setspace}

\babelprovide
  [
    , import
    , language = Chinese Simplified
  ]
  {chinese-simplified}

\babelfont [chinese-simplified] {sf}
  [
    , ItalicFont         = Noto Sans CJK SC DemiLight
    , BoldFont           = Noto Sans CJK SC Medium
    , BoldItalicFont     = Noto Sans CJK SC Medium
    , UprightFeatures    = {
                             , SmallCapsFont = {Noto Sans CJK SC DemiLight}
                           }
    , ItalicFeatures     = {
                             , FakeSlant     = 0.15
                             , SmallCapsFont = {Noto Sans CJK SC DemiLight}
                           }
    , BoldFeatures       = {
                             , SmallCapsFont = {Noto Sans CJK SC Medium}
                           }
    , BoldItalicFeatures = {
                             , FakeSlant     = 0.15
                             , SmallCapsFont = {Noto Sans CJK SC Medium}
                           }
  ]
  {Noto Sans CJK SC DemiLight}

\babelprehyphenation {chinese-simplified} { [‘“][‘“] }
  {
    {} ,
    { insert, penalty = 10000 },
    { insert, space = -.5 0 0 },
    {}
  }
\babelprehyphenation {chinese-simplified} { [’”][’”] }
  {
    {},
    { insert, penalty = 10000 },
    { insert, space = -.5 0 0 },
    {}
  }
\babelprehyphenation {chinese-simplified} { [？。！][’”] }
  {
    {},
    { insert, penalty = 10000 },
    { insert, space = -.4 0 0 },
    {}
  }
\babelprehyphenation {chinese-simplified} { [’”][？。！；] }
  {
    {},
    { insert, penalty = 10000 },
    { insert, space = -.4 0 0 },
    {}
  }
\babelprehyphenation {chinese-simplified} { [，。：][‘“] }
  {
    {},
    { insert, space = -.5 0 0 },
    {}
  }

\babelcharproperty{`“}{linebreak}{op}
\expandafter\addto\csname extraschinese-simplified\endcsname{\onehalfspacing}

\SetProtrusion
  {
    , encoding = TU
    , family   = Noto Sans CJK SC DemiLight
  }
  {
    , “  = { 600 , }
    , ”  = { , 600 }
    , 、 = { , 700 }
    , 。 = { , 700 }
    , 「 = { 450 , }
    , 」 = { , 450 }
    , ！ = { , 600 }
    , ， = { , 700 }
    , ： = { , 600 }
    , ； = { , 600 }
    , ？ = { , 500 }
  }

  {
    , language         = chinese-simplified
    , version          = CCB
    , selah/format     = (##1)
    , selah/text       = 细拉
    , poetry/indent    = 0em
    , poetry/bigindent = 2em
  }
